102  uint16_t CheckSum(Header *header, uint32_t size) {
103  uint8_t *buf = reinterpret_cast<uint8_t *>(header);
104  uint64_t sum = 0;
105 
106  while (size > 1) {
107  sum += *reinterpret_cast<uint16_t*>(buf);
108  buf += 2;
109  if(sum & 0x80000000) {
110  // if high order bit set, fold
111  sum = (sum & 0xFFFF) + (sum >> 16);
112  }
113  size -= 2;
114  }
115 
116  if (size) {
117  // take care of left over byte
118  sum += static_cast<uint16_t>(*buf);
119  }
120 
121  while (sum >> 16) {
122  sum = (sum & 0xFFFF) + (sum >> 16);
123  }
124 
125  return ~sum;
126  }
