56 template <typename T>
57 void Allocator<T>::Free(T *data) {
58  Locker locker(_lock);
59  Container *before = nullptr;
60  Container *cur = _list;
61  while(cur != nullptr) {
62  if (cur->_entry <= data && data <= cur->_entry + 63) {
63  if (~cur->_flag == 0 && before != nullptr) {
64  before->_next = cur->_next;
65  cur->_next = _list;
66  _list = cur;
67  }
68  assert(((reinterpret_cast<size_t>(data) - reinterpret_cast<size_t>(cur->_entry)) % sizeof(T)) == 0);
69  cur->_flag &= ~(1 << (data - cur->_entry));
70  return;
71  }
72  before = cur;
73  cur = cur->_next;
74  }
75  assert(false);
76 }

78 template <typename T>
79 T *Allocator<T>::Alloc() {
80  bool extend = false;
81  T* rval = nullptr;
82  while(true) {
83  // 最適化回避
84  volatile size_t *_list_ptr = const_cast<volatile size_t *>(reinterpret_cast<size_t *>(&this->_list));
85  while(~reinterpret_cast<Container *>(*_list_ptr)->_flag == 0) {
86  _list_ptr = const_cast<volatile size_t *>(reinterpret_cast<size_t *>(&this->_list));
87  asm volatile ("nop");
88  }
89  {
90  Locker locker(_lock);
91  int i;
92  for (i = 0; i < 64; i++) {
93  uint64_t bit = static_cast<uint64_t>(1) << i;
94  if ((_list->_flag & bit) == 0) {
95  break;
96  }
97  }
98  if (i == 64) {
99  break;
100  }
101  rval = &_list->_entry[i];
102  _list->_flag |= static_cast<uint64_t>(1) << i;
103  if (~_list->_flag == 0) {
104  if (_list->_next == nullptr || ~_list->_next->_flag == 0) {
105  extend = true;
106  break;
107  }
108  Container *tmp = _list;
109  _list = _list->_next;
110  Container *cur = _list;
111  while(cur->_next != nullptr && ~cur->_next->_flag != 0) {
112  cur = cur->_next;
113  }
114  tmp->_next = cur->_next;
115  cur->_next = tmp;
116  }
117  }
118  if (rval != nullptr) {
119  break;
120  }
121  }
122  if (extend) {
123  return Extend(rval);
124  }
125  return rval;
126 }
