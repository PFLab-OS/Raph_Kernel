79  // rxパケットの処理の流れ
80  // 0. rx_reservedを初期化、バッファを満タンにしておく
81  // 1. Receiveハンドラがパケットを受信すると、rx_reservedから一つ取り出し、
82  // memcpyの上、rx_bufferedに詰む
83  // 2. プロトコル・スタックはReceivePacket関数を呼ぶ
84  // 3. ReceivePacket関数はrx_bufferedからパケットを取得する
85  // 4. プロトコル・スタックは取得したパケットを処理した上でReuseRxBufferを呼ぶ
86  // 5. ReuseRxBufferはrx_reservedにバッファを返す
87  // 6. 1に戻る
88  //
89  // プロトコル・スタックがReuseRxBufferを呼ばないと
90  // そのうちrx_reservedが枯渇して、一切のパケットの受信ができなくなるるよ♪
91  NetDevRingBuffer _rx_reserved;
92  NetDevFunctionalRingBuffer _rx_buffered;
93 
94  // txパケットの処理の流れ
95  // 0. tx_reservedを初期化、バッファを満タンにしておく
96  // 1. プロトコル・スタックはGetTxBufferを呼び出す
97  // 2. GetTxBufferはtx_reservedからバッファを取得する
98  // 3. プロトコル・スタックはバッファにmemcpyして、TransmitPacket関数を呼ぶ
99  // 4. TransmitPacket関数はtx_bufferedにパケットを詰む
100  // 5. Transmitハンドラがパケットを処理した上でtx_reservedに返す
101  // 6. 1に戻る
102  //
103  // プロトコル・スタックはGetTxBufferで確保したバッファを必ずTransmitPacketするか
104  // ReuseTxBufferで開放しなければならない。サボるとそのうちtx_reservedが枯渇
105  // して、一切のパケットの送信ができなくなるよ♪
106  NetDevRingBuffer _tx_reserved;
