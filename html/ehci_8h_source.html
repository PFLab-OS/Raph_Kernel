118  void SetTokenAndBuffer(bool interrupt_on_complete, bool data_toggle, UsbCtrl::PacketIdentification pid, int total_bytes, virt_addr buf) {
119  SetTokenAndBufferSub(interrupt_on_complete, data_toggle, pid, total_bytes);
120  phys_addr buf_p = k2p(buf);
121  phys_addr buf_p_end = buf_p + total_bytes;
122  if (buf_p_end > 0xFFFFFFFF) {
123  kernel_panic("Ehci", "64-bit address is not supported");
124  }
125  _buffer_pointer[0] = buf_p;
126  buf_p = ((buf_p + 4096) / 4096) * 4096;
127  for (int i = 1; i < 5; i++) {
128  if (buf_p <= buf_p_end) {
129  _buffer_pointer[i] = buf_p;
130  buf_p += 4096;
131  } else {
132  _buffer_pointer[i] = 0;
133  }
134  }
135  }
