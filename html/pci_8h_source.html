90  int GetMsiCount(uint8_t bus, uint8_t device, uint8_t func) {
91  uint16_t offset = FindCapability(bus, device, func, CapabilityId::kMsi);
92  if (offset == 0) {
93  return 0;
94  }
95 
96  uint16_t control = ReadReg<uint16_t>(bus, device, func, offset + kMsiCapRegControl);
97  uint16_t cap = (control & kMsiCapRegControlMultiMsgCapMask) >> kMsiCapRegControlMultiMsgCapOffset;
98  int count = 1;
99  for (; cap > 0; cap--) {
100  count *= 2;
101  }
102  return count;
103  }
