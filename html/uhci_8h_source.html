350 inline void DevUhci::DisablePort(int port) {
351  while(true) {
352  WriteControllerReg<uint16_t>(kCtrlRegPortBase + port * 2, ReadControllerReg<uint16_t>(kCtrlRegPortBase + port * 2) & ~kCtrlRegPortFlagEnable);
353  if ((ReadControllerReg<uint16_t>(kCtrlRegPortBase + port * 2) & kCtrlRegPortFlagEnable) == 0) {
354  break;
355  }
356  asm volatile("":::"memory");
357  }
358 }

370 inline void DevUhci::ResetPort(int port) {
371  DisablePort(port);
372  WriteControllerReg<uint16_t>(kCtrlRegPortBase + port * 2, ReadControllerReg<uint16_t>(kCtrlRegPortBase + port * 2) | kCtrlRegPortFlagReset);
373  // set reset bit for 50ms
374  timer->BusyUwait(50*1000);
375  // then unset reset bit
376  WriteControllerReg<uint16_t>(kCtrlRegPortBase + port * 2, ReadControllerReg<uint16_t>(kCtrlRegPortBase + port * 2) & ~kCtrlRegPortFlagReset);
377  EnablePort(port);
378 }
