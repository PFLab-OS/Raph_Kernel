149 /*
150  * Allocate a device specific dma_tag encapsulating the constraints of
151  * the parent tag in addition to other restrictions specified:
152  *
153  * alignment: Alignment for segments.
154  * boundary: Boundary that segments cannot cross.
155  * lowaddr: Low restricted address that cannot appear in a mapping.
156  * highaddr: High restricted address that cannot appear in a mapping.
157  * filtfunc: An optional function to further test if an address
158  * within the range of lowaddr and highaddr cannot appear
159  * in a mapping.
160  * filtfuncarg: An argument that will be passed to filtfunc in addition
161  * to the address to test.
162  * maxsize: Maximum mapping size supported by this tag.
163  * nsegments: Number of discontinuities allowed in maps.
164  * maxsegsz: Maximum size of a segment in the map.
165  * flags: Bus DMA flags.
166  * lockfunc: An optional function to handle driver-defined lock
167  * operations.
168  * lockfuncarg: An argument that will be passed to lockfunc in addition
169  * to the lock operation.
170  * dmat: A pointer to set to a valid dma tag should the return
171  * value of this function indicate success.
172  */

288 /*
289  * Perform a synchronization operation on the given map. If the map
290  * is NULL we have a fully IO-coherent system. On every ARM architecture
291  * there must be a memory barrier placed to ensure that all data
292  * accesses are visible before going any further.
293  */
294 void _bus_dmamap_sync(bus_dma_tag_t, bus_dmamap_t, bus_dmasync_op_t);
295 #if defined(__arm__)
296  #define __BUS_DMAMAP_SYNC_DEFAULT mb()
297 #elif defined(__aarch64__)
298  #define __BUS_DMAMAP_SYNC_DEFAULT dmb(sy)
299 #else
300  #define __BUS_DMAMAP_SYNC_DEFAULT do {} while (0)
301 #endif
302 #define bus_dmamap_sync(dmat, dmamap, op) \
303  do { \
304  if ((dmamap) != NULL) \
305  _bus_dmamap_sync(dmat, dmamap, op); \
306  else \
307  __BUS_DMAMAP_SYNC_DEFAULT; \
308  } while (0)
